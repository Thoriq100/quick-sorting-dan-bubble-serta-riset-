int quick(int kiri, int kanan){

int pivot;
		
		if (kiri<kanan){
		i = kiri;
		j = kanan;
		pivot=data[j].kodebk;
	do{
		while (i<j &&data[i].kodebk<=pivot){
		i++;
		}
		while (j>i && data[j].kodebk>=pivot){
		j--;
		}
		if (i<j){

  		temp=data[i];
  		data[i]=data[j];
  		data[j]=temp;
	}
}while (i<j);

		temp =data[j];
		data[j]=data[kanan];
		data[kanan]=temp;
	
	if (j-kiri<kanan-i){

		quick(kiri,j-1);
		quick(i+1,kanan);
	}else{
		quick(i+1,kanan);
		quick(kiri,j-1);
		}
	}
}
